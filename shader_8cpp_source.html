00289     void Shader::apply(){
00290         if (shaderProgram){
00291             glDeleteShader(shaderProgram);
00292         }
00293         shaderProgram = glCreateProgram();
00294         vector<ShaderObj> shaderObjects;
00295         vector<ShaderType> requiredTypes{ShaderType::VertexShader, ShaderType::FragmentShader};
00296         for (auto & element : shaderSources) {
00297             auto shaderType = element.first;
00298             string & source = element.second;
00299             if (source.length()>0){
00300                 string precompiledSource = getPrecompiledSource(source);
00301                 
00302                 shaderObjects.push_back(compileShader(precompiledSource, shaderType));
00303                 // remove element from vector
00304                 requiredTypes.erase(std::remove(requiredTypes.begin(), requiredTypes.end(), shaderType), requiredTypes.end());
00305                 glAttachShader( shaderProgram,  shaderObjects.back());
00306             }
00307         }
00308         if (requiredTypes.size()>0){
00309             throw ShaderBuildException{"Both vertex and fragment shader required.", ShaderErrorType::IncompleteShader};
00310         }
00311         
00312         linkProgram();
00313         // shaderObjects deleted when goes out of scope (which is ok as long as program is not deleted)
00314         glUseProgram(shaderProgram);
00315 
00316         // clean up
00317         for (auto & shaderObject : shaderObjects){
00318             shaderObject.detach(shaderProgram);
00319         }
00320         shaderObjects.clear();
00321 
00322         shaderAttributes = getActiveShaderAttributes(shaderProgram);
00323         shaderUniforms = getActiveShaderUniforms(shaderProgram);
00324 
00325         updateDefaultShaderLocation();
00326 
00327         shaderChanged.notifyListeners({this, ShaderEventType::shader });
