00036 
00037             // compute half vector
00038             directionalLightData[2] = -normalize(vec3(0,0,-1) + directionalLightDirection);
00039 
00040             // set color intensity
00041             directionalLightData[1] = directionalLight->getColorIntensity();
00042         }  else {
00043             directionalLightData = mat3(0);
00044         }
00045         int i = 0;

00047             Light * light = pointLights[i];
00048             Transform * transform = light->getGameObject()->getTransform();
00049 
00050             // save eyespace position
00051             pointLightData[i][0] = (vec3)(viewMatrix * vec4(transform->getPosition(),1.0f));
00052             pointLightData[i][1] = light->getColorIntensity();
00053             pointLightData[i][2] = light->getAttenuation();
00054         }
00055         // reset unused
